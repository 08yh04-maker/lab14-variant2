FROM golang:1.21-alpine AS builder

WORKDIR /app

# Копируем go.mod и скачиваем зависимости
COPY go.mod go.sum ./
RUN go mod download

# Копируем исходный код
COPY . .

# Собираем статический бинарник
RUN CGO_ENABLED=0 GOOS=linux go build -a -installsuffix cgo -o collector main.go arrow_server.go validator.go

# Финальный образ
FROM alpine:latest

RUN apk --no-cache add ca-certificates

WORKDIR /root/

COPY --from=builder /app/collector .
COPY --from=builder /app/weather_validator.dll . 2>/dev/null || true

EXPOSE 50051

CMD ["./collector"]
